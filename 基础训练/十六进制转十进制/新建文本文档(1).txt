问题描述
　　从键盘输入一个不超过8位的正的十六进制数字符串，将它转换为正的十进制数后输出。
　　注：十六进制数中的10~15分别用大写的英文字母A、B、C、D、E、F表示。
样例输入
FFFF
样例输出
65535




#include "iostream"
#include "string"
using namespace std;
int main(){
	long long result=0;
	string s;
	cin>>s;
	int i;
	for(i=0;s[i]!='\0';i++){
	        if(s[i]>='A'&&s[i]<='F')
	        	        result=result*16+(s[i]- 'A')+10;  
	        	        if(s[i]>='0'&&s[i]<='9')
	        	         result=result*16+(s[i]- '0'); 
		} 
		cout<<result;
	return 0;
}
